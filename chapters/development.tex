[height width] = size(image);
imageCopy = repmat(uint8(0), height, width);
divider = 0.0;
maxLevel = double(usedGrayLevels);

while 1
    divider = divider + 0.01;
    if maxLevel/divider <= 255
        break;
    end
end

    for w=1:1:width
       imageCopy(h, w) = image(h, w)/divider;
    end
end
